### Approve 패턴

```python
# tests/golden_master/approve.py — excerpt
def read_expected(expected_path: Path) -> str:
    with expected_path.open(encoding="utf-8") as handle:
        return handle.read()

def approve(actual: str, expected_path: Path = DEFAULT_GOLDEN_MASTER_PATH) -> None:
    if not expected_path.exists():
        expected_path.write_text(actual, encoding="utf-8")
        pytest.fail("Golden Master baseline created ... Re-run tests ...")
    expected = read_expected(expected_path)
    if actual != expected:
        diff_text = format_unified_diff(expected, actual)  # --- expected / +++ actual
        pytest.fail(f"Golden Master mismatch.\n{diff_text}")
```
